\section{Введение}
При исследовании нормальных расширений без высшего ветвления возникла задача описания всех неразветвленных расширений локального поля, которые вместе с основным полем являются  {\it регулярными}, то есть не содержащими нетривиальных $p$-х корней из $1$ (где $p$ --- характеристика поля вычетов локального поля). Основное локальное поле при этом называется {\it вполне регулярным}. Эта задача была решена З. И. Боревичем в работе \cite{B}. Основной результат этой работы, доказательство которого было упрощено Д. К. Фаддеевым, является следующим.

\begin{thm} \label{th1}
	 Пусть $K$ --- регулярное локальное поле. Расширение $K(\zeta)/K$, где $\zeta^p = 1$, $\zeta \neq 1$, будет неразветвленным тогда итолько тогда, когда индекс ветвления \linebreak $e = e(K/\Qp)$ делится на $p-1$.
\end{thm}

Встает естественный вопрос --- является ли ограничение на индекс ветвления исходного поля, которое не является регулярным, то есть содержит нетривиальный корень степени $p$ из $1$, достаточным для того, чтобы круговое расширение $K(\zeta_{p^n})$, $n \Geq 2$, было бы неразветвлено. Этот вопрос решается в первой части работы.

Задача о регулярных и вполне регулярных полях возникает в арифметике формальных модулей и она решается во второй части настоящей работы.

\section{Вполне разветвленное $K(\zeta_{p^2})/K$ \\ при нерегулярном $K$ \\ с индексом ветвления $p(p-1)$}
$K$ --- локальное поле (конечное расширение $\Qp$),\\
$\zeta_{p^m}$ --- первообразный корень степени $p^m$ из $1$,\\
$e$ --- абсолютный индекс ветвления поля $K$,\\
$\Teich$ --- мультипликативная система Тейхмюллера в поле $K$,\\
$\ORing$ --- кольцо целых подполя инерции $T$ в $K/\Qp$,\\
$E(f(X)) = exp(1 + {\Delta \over p} + {\Delta^2 \over {p^2}} + \ldots)(f(X))$, где $\Delta f(X) = f(X^p)$, для $f(X) \in \Zp[[X]]X$.
Считаем, что $\zp \in K$ и индекс ветвления $e$ делится на $p$. 

Докажем, что существуют поля, для которых расширение $K(\zpp)$ будет вполне разветвленным над $K$.

Пусть $K_0 = \Qp(\zp)$ и $K = K_0(\sqrt[p]{\pi})$, где $\pi = 1-\zp$. Пусть $\Pi = \sqrt[p]{\pi}$.
\begin{thm}
Расширение $K(\zpp)/K$ вполне разветвлено.
\end{thm}
\begin{proof}
В нашем случае $e = e(K/\Qp) = p(p-1)$. Рассмотрим разложение корня $\zp$ по образующим мультипликативной группы $K^*$. Пусть $a \in \Zp$ и $\omega(a) = \linebreak = E(a(\underline{\zeta}^p - 1)) \vert_{X = \Pi}$, где $\underline{\zeta}(X) = (1-X^p)^p$.

В работе \cite[\S 4, предложение 6]{V} было доказано, что $\omega(a)$ --- $p$-примарный элемент поля $K$ (то есть расширение $K(\sqrt[p]{\omega(a)})/K$ неразветвлено), и при этом значение символа Гильберта в поле $K$, $(,)_p: K^* \times K^* \rightarrow \langle\zp\rangle$ на паре $\Pi, \omega(a)$ равно $(\Pi, \omega(a)) = \zp^a$.

Образующими $K^*$ будут элементы $\{ \Pi, \omega(a), 1 - \theta \pi^b \mid \linebreak \theta \in \Teich, 1 \Leq b < p^2, p \nmid b \}$, и корень $\zp$, тем самым, раскладывается в виде
\begin{equation} \label{eq:product}
\zp = \omega(a)^\beta \prod\limits_{\substack{p \nmid b \\ 1 \Leq b < p^2}} (1 - \theta \Pi^b)^{\alpha_{\theta,b}},
\end{equation}
где $\beta, \alpha_{\theta,b} \in \mathbb{Z}$.

Докажем сперва, что 
\begin{equation} \label{eq:beta}
\beta \equiv 0 \mod p.
\end{equation}
Для этого подсчитаем значение символа Гильберта на паре $\Pi, \zp$. Нетрудно видеть, что из соотношения Стейнберга $(\alpha, 1 - \alpha) = 1$, $\alpha \neq 0$ для символа Гильберта следует равенство
\begin{equation} \label{eq:hilb}
(\Pi, 1 - \theta \Pi^b) = 1, p \nmid b.
\end{equation}
Действительно, $1 = (\theta \Pi^b, 1 - \theta \Pi^b) = (\theta, 1 - \theta \Pi^b) \cdot (\Pi, 1 - \theta \Pi^b)^b$. При этом $\theta = \theta_1^p$ при некотором $\theta_1 \in \Teich$, так как группа $\Teich$ $p$-делима. Значит, $(\Pi, 1 - \theta \Pi^b)^b = 1$, откуда $(\Pi, 1 - \theta \Pi^b) = 1$, так как, если $(\Pi, 1 - \theta \Pi^b) = \zeta^k$ при некотором $1 \Leq k \Leq p-1$, то $\zeta^{bk} = 1$, что противоречит $p \nmid bk$.

Из равенства \eqref{eq:hilb} следует 
$$(\Pi,\zp) = (\Pi,\omega(a))^\beta \cdot \prod\limits_{\substack{p \nmid b \\ 1 \Leq b < p^2}} (1 - \theta \Pi^b)^{\alpha_{\theta,b}} =$$
$$= (\Pi,\omega(a))^\beta = \zp^{a\beta}, \text{то есть}$$
\begin{equation} \label{eq:hilbzeta}
(\Pi,\zp) = \zp^{a\beta}.
\end{equation}

Подсчитаем теперь значение $(\Pi,\zp)$ по явной формуле для символа Гильберта (см. \cite[(12)]{V}). Обозначим через $l(1 + f(X)))$ обратную функцию к функции Артина-Хассе $E(f(X))$. Она была определена в \cite[\S 1, п. 1]{V}:
$$l(1 + f(X)) = (1 - {\Delta \over p})\log(1 + f(X))$$
для $f(X) \in X\Zp[[X]]$ и $\Delta f(X) = f(X^p)$.

Тогда имеет место формула 
$$(\Pi,\zp) = \zp^{\res_X X^{-1} l(\underline{\zeta}) / ((1 - \underline{\zeta}^p)^p - 1)}.$$
Вычислим значение $\res_X X^{-1} l(\underline{\zeta}) / ((1 - \underline{\zeta}^p)^p - 1) \mod p$ для $\underline{\zeta} = \underline{\zeta}(X) = 1 - X^p$. Ясно, что
\begin{equation} \label{eq:resmodp}
(1 - \underline{\zeta}^p)^p - 1 \equiv -\underline{\zeta}^{p^2} \mod p.
\end{equation}
Кроме того, 
$$ l(\underline{\zeta}) = (1 - {\Delta \over p}) \log(1 - X^p) = $$
$$ = \sum\limits_{p \nmid m}{X^{pm} \over m} + \sum\limits_{m \Geq 1}{X^{p^{m+1}} - X^{p^{m+1}} \over p^m} = \sum\limits_{p \nmid m}{X^{pm} \over m}.$$
Среди степеней $X^{pm}$, $p \nmid m$ нет степени $p^2$, поэтому 
$$\res_X X^{-1} l(\underline{\zeta}) / -\underline{\zeta}^{p^2} \equiv 0 \mod p,$$
значит \linebreak $(\Pi, \zp) = 1$. Отсюда и из \eqref{eq:hilbzeta} следует \eqref{eq:beta}.

Значит $\zp \equiv \prod\limits_{\substack{p \nmid b \\ 1 \Leq b < p^2}} (1 - \theta \Pi^b)^{\alpha_{\theta, b}} \mod K^{*p}$. Пусть степень $b$ --- наименьшая, для которой $\alpha_{\theta, b} \not\equiv 0 \mod p$. Если такой нет, то это означает, что $\zp \in K^{*p}$, что невозможно. Тогда $K(\zpp) = K(\sqrt[p]{\zp}) = K(\sqrt[p]{\varepsilon})$, где $\varepsilon = 1 - c\Pi^b$, а $c$ --- некоторая единица поля $K$, то есть $c = c_0 + c_1\Pi + c_2\Pi^2 + \ldots$, $c_i \in \Zp$ и $p \nmid c_0$, $1 \Leq b < {pe \over p-1}$, $p \nmid b$.

Таким образом $K(\zpp)$ получается присоединением корня уравнения $X^p = \varepsilon$. Заменим переменную $X = Y + 1$, тогда $Y^p + C_p^{p-1}Y^{p-1} + \cdots + pY = -c\Pi^b$. Пусть $\chi$ --- корень последнего уравнения. Предположим, что $K(\zpp)/K$ неразветвлено, тогда возможны три случая для нормирования $\val(\chi) = \val_{K(\zpp)}(\chi)$.
\begin{enumerate}
	\item $\val(\chi) = 0$, тогда $\val(\chi^p + \cdots + p\chi) = 0$, но это невозможно, так как $\val(-c\Pi^b) = b > 0$.
	\item $1 \Leq \val(\chi) \Leq {e \over p-1}$, тогда $\val(\chi^p + \cdots + p\chi) = \val(\chi^p) = \linebreak = p\val(\chi)$ делится на $p$, но это невозможно, так как $\val(-c\Pi^b) = b$ не делится на $p$.
	\item $\val(\chi) > {e \over p-1}$, тогда $\val(\chi^p + \cdots + p\chi) \Geq {pe \over p-1}$, так как $\val(\chi^p) = p\val(\chi) > {pe \over p-1}$ и $\val(p\chi) = e + \val(\chi) = {pe \over p-1}$. Но это невозможно, так как $\val(-c\Pi^b) = b < {pe \over p-1}$.
\end{enumerate}
Мы предположили неразветвленность расширения \linebreak $K(\zpp)/K$ и получили противоречие.

Отсюда вытекает, что $K(\zpp)/K$ вполне разветвлено.
\end{proof}

\section{Формальные модули в \\ многомерных локальных полях}
В этом разделе формальный модуль будет строиться только для формальной группы Хонды высоты $h$.\\
Пусть поле $K$ --- полное многомерное ($n$-мерное) поле нулевой характеристики, то есть поле, для которого имеется последовательность полей, $$K_0, K_1, \ldots, K_{n-1}, K_n = K$$ таких, что $K_0$ --- совершенное поле характеристики $p$, а $K_{i-1}$ --- поле вычетов для $K_i, 1\Leq i\Leq n.$\\
В дальнейшем мы предполагаем, что последнее поле вычетов конечно. Также, следуя изложению работы \cite{Inv}, мы будем рассматривать только {\it стандартные} многомерные поля для интересующего нас разнохарактеристического случая $n$-мерного локального поля (то есть $charK=0$,\linebreak $charK_{n-1}=p>0$). Последовательность полей тогда выглядит следующим образом: \pagebreak
$$\mathbb{F}_q - \mathbb{F}_q((t_1)) - \ldots - \mathbb{F}_q((t_1))\ldots((t_{n-1})) -$$
$$ - k\{\{t_1\}\}\ldots\{\{t_{n-1}\}\}:=K,$$ где 
$k$ --- конечное расширение поля $\mathbb{Q}_p$, а $q=\#\overline{k}.$\\

\subsection{Случай двумерного локального поля для формальной группы Хонды} \label{dva}
Чтобы не усложнять процедуру доказательства общей теоремы техническими выкладками, докажем её для двумерного поля $$\mathbb{F}_p - \mathbb{F}_p((t_0)) - k\{\{t_0\}\}:=K,$$ %с системой локальных параметров $\{t_0,p\}$.
где $k$ --- конечное расширение $\mathbb{Q}_p$.\\ Пусть $T$ --- подполе инерции в $k$, $T':=T\{\{t_0\}\}$, \linebreak$\oo_{T'}$ --- его кольцо целых относительно двумерного нормирования. Тогда имеем такую схему двумерных полей:
$$
\begin{CD}
\mathbb{Q}_p\{\{t_0\}\} @>>> T\{\{t_0\}\}=T' @>>> k\{\{t_0\}\} = K\\
\mathbb{Z}_p\{\{t_0\}\} @>>> \oo_{T'} @>>> \oo_K\\
\{t_0,p\} @>>> \{t_0,p\} @>>>\{t_0,\pi\}
\end{CD}
$$
Определим формальнцую группу Хонды над $\oo_{T'}$. Для этого фиксируем эндоморфизм $\sigma:\oo_{T'} \rightarrow \oo_{T'}$, определенный следующим образом: $$\sigma(\sum a_it^i_0)=\sum a_i^{Fr}t^{p \cdot i}_0, ~ a_i \in \oo_T,$$ и обладающий свойством:
$$\forall x \in \oo_{T'} \quad \sigma(x) \equiv x^p \pmod {\mm_{T'}},$$
где $Fr$ --- продолжение автоморфизма Фробениуса на $T/\mathbb{Q}_p$. Определим действие оператора Фробениуса на рядах из $\oo_T'[|X|]$
следующим образом: $$\bigtriangleup(\sum b_ix^i)=\sum \sigma(b_i)x^{p\cdot i}, ~ b_i \in \oo_{T'}.$$
Множество операторов $\sum\limits_{i\Geq 0} b_i \bigtriangleup^i$, где $b_i \in \oo_{T'}$, образуют некоммутативное кольцо $\oo_{T'}[|\bigtriangleup|]$ формальных степенных рядов от $\bigtriangleup$, для которого $\bigtriangleup b = \sigma(b)\bigtriangleup$ для $b \in \oo_{T'}$. Формальная группа Хонды над $\oo_{T'}$ с логарифмом \linebreak$\log_F(X) \in T'[|X|]$ задается также, как и в параграфе \ref{Xon}.\\
Пусть теперь $F(X,Y)$ --- формальная группа Хонды над $\oo_{T'}$ высоты $h$, $\End_{\oo_K}F \cong \mathbb{Z}_p\{\{t_0\}\}$. Пусть поле $L$ --- расширение поля $K$ такое, что $\Ker_F[p]$ (без нуля) не содержится в $L$, $e=e(L/T') = (e_1,e_2)$ --- $2$-мерный индекс ветвления, $\{T_0,\Pi\}$ --- система локальных параметров поля $L$. Тогда $T_0$ и $\Pi$ являются делителями $t_0$ и $\pi$ соответственно. Диаграмма полей в этом случае выглядит так:
$$
\begin{CD}
\mathbb{Q}_p\{\{t_0\}\} @>>> T' @>>> K @>>> L\\
\mathbb{Z}_p\{\{t_0\}\} @>>> \oo_{T'} @>>> \oo_K @>>> \oo_L\\
\{t_0,p\} @>>> \{t_0,p\} @>>>\{t_0,\pi\} @>>>\{T_0,\Pi\}
\end{CD}
$$
\begin{thm} \label{2d} Поле $L$ (а значит и $F(\mm_L)$) являются вполне регулярными относительно формальной группы $F$ тогда и только тогда, когда первый индекс $e_1$ в $2$-мерном индексе ветвления $e=(e_1,e_2)$ расширения $L/T'$ не делится на\linebreak $p^h-1$. \end{thm}
\begin{proof}
Из определения высоты формальной группы следует, что $$[p]_F(X)\equiv cX^{p^h} \mod (\pi_0, deg(p^h+1)), \quad c\in \oo_{T'}^*,$$ при этом $c$ является рядом $$c=c_0+c_1\pi+\ldots, \ ~c_i\in \mathbb{Z}_p\{\{t_0\}\}, ~ c_0 \in \mathbb{Z}_p\{\{t_0\}\}^*\cap \oo_{T'}^*,$$ то есть его коэффициенты сами являются рядами с коэффициентами из $\mathbb{Z}_p$. Однако обратимость $c_0$ в $\oo_{T'}$ выполнена только в том случае, когда $c_0$ начинается со свободного коэффициента. Действительно, пусть $$c_0=a_mt^m+a_{m+1}t^{m+1}+\ldots, ~ m\in \mathbb{Z}, a_i \in \mathbb{Z}_p, a_m \in \mathbb{Z}_p^*$$
Пусть $a^{'}_{m+k}=\dfrac{a_{m+k}}{a_m}.$ Тогда $$c_0=a_mt^m(1+a^{'}_{m+1}t^{m+1}+\ldots),$$ откуда
$$c^{-1}_0=a^{-1}_mt^{-m}(1+a^{'}_{m+1}t^{m+1}+\ldots)^{-1}.$$ Этот ряд лежит в $\mathbb{Z}_p\{\{t_0\}\}^*$, но не лежит в $\oo_{T'}^*$ и даже $\oo_{T'}$, поскольку $$\oo_K=\{\alpha \mid \val_K(\alpha)\Geq(0,0)\},$$ а $\val_K(t^{-m})=(-m,0)<(0,0)$. Таким образом, $m=0$ и тогда $$c=(\theta_{0,0}+\theta_{0,1}t+\ldots) + (\theta_{1,0}t^{-m_1}+\ldots)\pi + (\theta_{2,0}t^{-m_2}+\ldots)\pi^2+\ldots,$$ где $\theta_{0,0} \in \mathbb{Z}_p^*, \theta_{i,j} \in \mathbb{Z}_p$.\\
Далее действуем аналогично случаю одномерного поля. Переобозначим для удобства $K_0 := \mathbb{Q}_p\{\{t_0\}\}$.

По подготовительной лемме Вейерштрасса существует $\varepsilon(X) \in 1+X\oo_{K_0}[|X|]$ такой, что 
$$f(x) := \dfrac{[p](X)}{X} \varepsilon(X) = $$
$$= p+pd_1X^1+\ldots+pd_{p^h-2}X^{p^h-2}+d_{p^h-1}X^{p^h-1},$$ где $d_{p^h-1}:=d_{p^h-1}(X) \in \oo_{T'}^*.$\\
Ясно, что $$\Ker[p]_F(X)=\Ker f(X).$$
Далее, сводим присоединение корня $\pi$ многочлена $f(X)$ к присоединению корня $\pi'$ уравнения $X^{p^h-1}+p'=0$, то есть имеем $$L(\Ker[p]_F)=L(\pi'),$$ где $p'=p\cdot d^{-1}_{p^h-1}$ --- простой в $K_0$.
\begin{enumerate}
\item Пусть первый индекс $e_1$ в двумерном индексе ветвления $(e_1,e_2)=e=e(L/T')=e(L/K_0)$ делится на $p^h-1$, то есть $e_1=(p^h-1)e',$ и пусть $\Pi$ --- простой элемент в $L$. Тогда $$-p'=\Pi^e\theta\eta,$$ где $\theta$ --- представитель Тейхмюллера, а $\eta$ --- главная единица поля $L$. Тогда $$\pi'=\sqrt[p^h-1]{-p'} = \Pi^{e'}\sqrt[p^h-1]{\theta}\eta_1,$$ где $\eta_1^{p^h-1}=\eta$ в $L$.\\
Поэтому расширение $L(\pi')=L(\sqrt[p^h-1]{\theta})$, а с ним и $L(\Ker[p]_F)$ неразветвлено над $L$. Это значит, что $F(\mm_L)$ не является вполне регулярным модулем.
\item Если же $e_1$ не делится на $p^h-1$, то расширение\linebreak $L(\pi')=L(\sqrt[p^h-1]{\pi^e\theta})$ будет разветвлено над $L$, откуда $F(\mm_L)$ --- вполне регулярный модуль.
\end{enumerate}
\end{proof}

\subsection{Случай многомерного локального \\ поля для формальной группы Хонды}
В этом параграфе мы сформулируем общий результат для многомерных локальных полей без доказательства, чтобы не загружать работу техническими выкладками. Идейно оно ничем не отличается от двумерного случая.\\
Пусть $k$ --- конечное расширение $\mathbb{Q}_p$, $T$ --- подполе инерции в $k$. Рассмотрим стандартное $n$-мерное локальное поле $K$: 
$$\mathbb{F}_q - \mathbb{F}_q((t_1)) - \ldots - \mathbb{F}_q((t_1))\ldots((t_{n-1})) - $$
$$ - k\{\{t_1\}\}\ldots\{\{t_{n-1}\}\}:=K.$$
Переобозначим для удобства $\mathbb{Q}'_p:=\mathbb{Q}_p\{\{t_1\}\}\ldots\{\{t_{n-1}\}\},$ \\ $T':=T\{\{t_1\}\}\ldots\{\{t_{n-1}\}\}, \mathbb{Z}'_p:= \mathbb{Z}_p\{\{t_1\}\}\ldots\{\{t_{n-1}\}\}$.\\ Фиксируем эндоморфизм $\sigma:\oo_{T'} \rightarrow \oo_{T'}$, определенный следующим образом: $$\sigma(\sum a_it^{i_1}_1\ldots t^{i_{n-1}}_{n-1})=\sum a_i^{Fr}t^{p \cdot i_1}_1\ldots t^{p \cdot i_{n-1}}_{n-1}, ~ a_i \in \oo_T,$$ и обладающий свойством:
$$\forall x \in \oo_{T'} \quad \sigma(x) \equiv x^p \pmod {\mm_{T'}},$$
где $Fr$ --- продолжение автоморфизма Фробениуса на $T/\mathbb{Q}_p$. Оператор Фробениуса и формальную группу Хонды определим также, как и в двумерном случае. %параграфе \ref{dva}.
Рассмотрим формальную группу Хонды $F(X,Y)$ над $\oo_{T'}$ высоты $h$, \linebreak $\End_{\oo_K}F \cong \mathbb{Z}'_p$.
Пусть $L$ --- расширение поля $K$ такое, что $\Ker_F[p]$ (без нуля) не содержится в $L$, $e=e(L/T') =\linebreak= (e_1,e_2,\ldots,e_n)$ --- $n$-мерный индекс ветвления, $\mm_L$ --- максимальный идеал кольца целых поля $L$, $F(\mm_L)$ --- формальный $\oo_L$-модуль на идеале $\mm_L$.\\
Пусть $\{T_1,\ldots,T_{n-1},T_n=\Pi\}$ --- система локальных параметров поля $L$.\\
Диаграмма полей будет иметь следующий вид:
$$
\begin{CD}
\mathbb{Q}'_p@>>> T' @>>> K @>>> L\\
\mathbb{Z}'_p @>>> \oo_{T'} @>>> \oo_K @>>> \oo_L\\
\end{CD}
$$

\begin{thm} \label{main} Поле $L$ (а значит и $F(\mm_L)$) являются вполне регулярными относительно формальной группы $F$ тогда и только тогда, когда первый индекс $e_1$ в $n$-мерном индексе ветвления $e=(e_1,\ldots, e_n)$ расширения $L/T'$ не делится на $p^h-1$. \end{thm}

\section{Формальные модули в многомерном поле}
\subsection{Формулировка}
Пусть $K$ --- полное многомерное поле нулевой характеристики, то есть поле, для которого имеется последовательнось полей, $K_0, K_1, \ldots, K_{n-1}, K_n = K$ таких, что $K_0$ --- совершенное поле характеристики $p$, и $K_{i-1}$ --- поле вычетов для $K_i$, $i \Leq i \Leq n$. \\
$(t_1, t_2, \ldots, t_{n-1}, t_n = \pi)$ --- система локальных параметров поля $K$, \\
$\ORing$ --- кольцо целых поля $K$ относительно $n$-мерного нормирования, \\
$F(X,Y)$ --- одномерная формальная группа над $\ORing$. \\
Будем предполагать, что $K$ --- разнохарактеристическое полное многомерное поле, то есть $char K = 0$, \linebreak $char K_{n-1} = p > 0$.

Пусть $k$ --- максимальное локальное поле в $K$ (конечное расширение $\Qp$), для которого поле \linebreak $E = k\{\{t_1\}\} \cdots \{\{t_{n-1}\}\} ((\pi))$ содерижится в $K$. \\
Предполагаем, что $K/E$ --- конечное расширение, и кольцо эндоморфизмов $\End_\ORing F$ формальной группы $F$ изоморфно кольцу целых $\ORing_0$ подрасширения $E_0$ в $E$. Считаем при этом, что $K/E_0$ --- конечное расширение.

Пусть $(t_1^{(0)}, t_2^{(0)}, \ldots t_{n-1}^{(0)}, t_n^{(0)} = \pi_0)$ --- система локальных параметров поля $E_0$, $\pi \in k_0$. \\
$E_0 = k_0\{\{t_1^{(0)}\}\} \cdots \{\{t_{n-1}^{(0)}\}\} ((\pi_0))$, причем $k_0$ --- подполе $k$. Пусть, далее $L$ --- расширение поля $K$ с системой локальных параметров $(T_1, T_2, \ldots, T_{n-1}, T_n = \Pi)$, и
\begin{equation} \label{eq:pi0andPi}
-\pi_0=T_1^{i_1} \cdots T_{n-1}^{i_{n-1}} \Pi^{e_n} \rho u, \tag{*}
\end{equation}
где $\rho$ --- представитель Тейхмюллера в $L$,  $u$ --- главная единица поля $L$.

Предполагаем, что поле $L$ --- регулярно относительно формальной группы $F$, то есть ядро изогении $\Ker[\pi_0]$ не содрежится в $L$ за исключением нуля. \\
$h := ht F$ --- высотра формальной группы $F$, \\
$\MIdeal := \ML$ --- максимальный идеал кольца целых поля $L$, \\
$F(\MIdeal)$ --- формальный $\ORing$-модуль на идеале $\MIdeal$. 

Будем называть поле $L$ вполне регулярным относительно формальной группы $F$, если $\Ker[\pi_0]$ не содержится в любом неразветвтленном расширении $M$ поля $L$.

В настоящей работе доказывается следующий результат

\begin{thm} \label{th:regularity}
	Поле $L$ является вполне регулярным относительно формальной группы $F$ тогда и только тогда, когда система индексов $(i_1, \ldots, i_{n-1}, e_n)$ делится на $p^h - 1$, то есть $i_k \vdots p^h - 1$, $1 \Leq k \Leq n$ (см. \eqref{eq:pi0andPi}).
\end{thm}

\subsection{Вспомогательные результаты}
Пусть $\xi \in \Ker[\pi_0]$, $\xi \neq 0$, $\xi \in K^{alg}$ --- произвольный элемент ядра изогении $[\pi_0]$. $[\pi_0](X) = \pi_0X\varepsilon(X) + X^{p^h}\theta r(X)$, где $\varepsilon(X), r(X) \in \ORing_0[[X]]$, при этом $\varepsilon(X) \equiv r(X) \equiv 1 \mod X$, а $\theta$ --- представитель система Тейхмюллера в кольце целых поля $k_0$. Ясно, что 
$${[\pi_0](X) \over X} = \pi_0 \varepsilon_1(X)^{p^h-1} + (X r_1(X))^{p^h-1}\theta,$$
где $\varepsilon_1(X)^{p^h-1} = \varepsilon(X)$, $r_1(X)^{p^h-1} = r(X)$. Поэтому элемент $\xi$ удовлетворяет уравнению $-\pi_0 = (\xi \ae(\xi))^{p^h-1} \theta$, где $\ae(X) = \varepsilon_1(X)^{-1} r_1(X)$. Поэтому
\begin{equation} \label{eq:Lxi}
L(\xi) = L(\sqrt[p^h - 1]{-\theta^{-1}\pi_0}).
\end{equation}

Пусть $e = e(K/\Qp)$; $1 \Leq b < {pe \over p-1}$, $p \nmid b$, $\Pi$ --- простой элемент в $K$, $\varepsilon = 1 + c\Pi^b$, $c$ --- единица в $K$.
\begin{lem}
	Пусть $\zeta_p \in K$, тогда $K(\sqrt[p]{\varepsilon}) / K$ вполне разветвленно.
\end{lem}
\begin{proof}
	Рассмотрим уравнения $X^p = \varepsilon$ и \linebreak $X = Y + 1$, тогда
	\begin{equation} \label{eq:Lxi}
	Y^p + C_p^{p-1}Y^{p-1} + \cdots + pY = c \Pi^b \tag{*}.
	\end{equation}
	Пусть расширение $K(\sqrt[p]{\varepsilon})/K$ неразветвлено. Тогда, если корень уравнения \eqref{eq:Lxi} $\alpha$ --- единица, то получим противоречие: $\val_L(\alpha) = 0 \implies \val(\alpha^p + \ldots + p\alpha) = 0$, но $\val(c\Pi^b) = b \Geq 1$. Здесь и далее $\val = \val_L$ --- нормирование в поле $L = K(\sqrt[p]{\varepsilon})$.\\
	Если $\val(\alpha) \Geq 1$, то есть два варианта.
	\begin{enumerate}
		\item $1 \Leq \val(\alpha) \Leq {e \over p-1}$, тогда $\val(\alpha^p + C_p^{p-1}\alpha^{p-1} + \cdots + p\alpha) = p\val(\alpha)$. Но $\val(c\Pi^b) = b \not\equiv 0 \mod p$ --- противоречие.
		\item $\val(\alpha) > {e \over p-1}$, тогда $\val(\alpha^p + C_p^{p-1}\alpha^{p-1} + \cdots + p\alpha) \Geq {pe \over p-1}$. С другой стороны, $\val(c\Pi^b) = b < {pe \over p-1}$.
	\end{enumerate}
	Итак, предположение, что $L/K$ неразветвлено приводит к противоречию.
\end{proof}
\begin{cor}
	Расширение $K(\zpp)/K$ вполне разветвлено.
\end{cor}
\subsection{Доказательство теоремы \eqref{th:regularity}}
\begin{proof}
		Пусть $\overline{e} = (e_1, \ldots, e_n)$ делится на $p^h-1$. Тогда
		$$ -\pi_0 = (T_1^{i_1'} \cdots T_{n-1}^{i_{n-1}'} \rho_1 u_1)^{p^h-1},$$
		где $i_k' = {i_k \over p^h-1}$, $1 \Leq k \Leq n-1$; $\rho_1^{p^h-1} = \rho$, $\rho \in \Teich$; и $u_1^{p^h-1} = u$, где $u_1$ --- главная единица в $L$. Отсюда и из \eqref{eq:Lxi} получаем $L(\xi) = L(\sqrt[p^h-1]{\theta^{-1}})$ и, значит, расширение $L/K$ неразветвлено.
		
		Если же $(i_1, \ldots, i_{n-1}, e_n)$ не делится на $p^h-1$, то \linebreak $L(\xi)=L(\sqrt[p^h-1]{T_1^{i_1} \cdots T_{n-1}^{i_{n-1}} \Pi^{e_n}})$ и $L(\xi)/L$ поэтому разветвлено.
\end{proof}
